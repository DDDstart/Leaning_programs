def update_team_stats(team, games=0, wins=0, draws=0, losses=0, points=0):
    if team not in teams:
        teams[team] = [0, 0, 0, 0, 0]  # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º: –∏–≥—Ä—ã, –ø–æ–±–µ–¥—ã, –Ω–∏—á—å–∏, –ø–æ—Ä–∞–∂–µ–Ω–∏—è, –æ—á–∫–∏
    teams[team][0] += games
    teams[team][1] += wins
    teams[team][2] += draws
    teams[team][3] += losses
    teams[team][4] += points

# –ß—Ç–µ–Ω–∏–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –º–∞—Ç—á–µ–π

    match = input().split(';')
    team1, goals1, team2, goals2 = match[0], int(match[1]), match[2], int(match[3])

    elif goals1 < goals2:
        # –ü–æ–±–µ–¥–∞ –≤—Ç–æ—Ä–æ–π –∫–æ–º–∞–Ω–¥—ã
        update_team_stats(team2, wins=1, points=3)
        update_team_stats(team1, losses=1)
    else:
        # –ù–∏—á—å—è
        update_team_stats(team1, draws=1, points=1)
        update_team_stats(team2, draws=1, points=1)

def substitution_cipher():
    # –ß—Ç–µ–Ω–∏–µ –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
    source_alphabet = input()
    target_alphabet = input()
    to_encrypt = input()
    to_decrypt = input()
    
    # –°–æ–∑–¥–∞–Ω–∏–µ —Å–ª–æ–≤–∞—Ä–µ–π —à–∏—Ñ—Ä–æ–≤–∞–Ω–∏—è –∏ —Ä–∞—Å—à–∏—Ñ—Ä–æ–≤–∫–∏
    encrypt_dict = {source_alphabet[i]: target_alphabet[i] for i in range(len(source_alphabet))}
    decrypt_dict = {target_alphabet[i]: source_alphabet[i] for i in range(len(target_alphabet))}
    
    # –®–∏—Ñ—Ä–æ–≤–∞–Ω–∏–µ —Å—Ç—Ä–æ–∫–∏
    encrypted_message = ''.join([encrypt_dict[char] for char in to_encrypt])
    
    # –†–∞—Å—à–∏—Ñ—Ä–æ–≤–∫–∞ —Å—Ç—Ä–æ–∫–∏
    decrypted_message = ''.join([decrypt_dict[char] for char in to_decrypt])
    
    # –í—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
    print(encrypted_message)
    print(decrypted_message)

def turtle_movement():
    # –ß—Ç–µ–Ω–∏–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –∫–æ–º–∞–Ω–¥
    n = int(input())

    x, y = 0, 0
    # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥
    for _ in range(n):
        command = input().split()
        direction = command[0]
        distance = int(command[1])
        # –ò–∑–º–µ–Ω–µ–Ω–∏–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è
        if direction == '—Å–µ–≤–µ—Ä':
            y += distance
        elif direction == '—é–≥':
            y -= distance
        elif direction == '–≤–æ—Å—Ç–æ–∫':
            x += distance
        elif direction == '–∑–∞–ø–∞–¥':
            x -= distance
    # –í—ã–≤–æ–¥ –∫–æ–Ω–µ—á–Ω—ã—Ö –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç
    print(x, y)

def calculate_average_height(file_name):
    # –°–æ–∑–¥–∞–µ–º —Å–ª–æ–≤–∞—Ä—å, –≥–¥–µ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞ –±—É–¥–µ—Ç —Å–ø–∏—Å–æ–∫ —Ä–æ—Å—Ç–æ–≤
    class_heights = {i: [] for i in range(1, 12)}
    # –ß—Ç–µ–Ω–∏–µ —Ñ–∞–π–ª–∞
    with open(file_name, 'r', encoding='utf-8') as file:
        for line in file:
            class_num, surname, height = line.strip().split('\t')
            class_num = int(class_num)  # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –Ω–æ–º–µ—Ä –∫–ª–∞—Å—Å–∞ –≤ —á–∏—Å–ª–æ
            height = int(height)        # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Ä–æ—Å—Ç –≤ —á–∏—Å–ª–æ
            class_heights[class_num].append(height)
    # –í—ã–≤–æ–¥ —Å—Ä–µ–¥–Ω–µ–≥–æ —Ä–æ—Å—Ç–∞ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞
    for class_num in range(1, 12):

            average_height = sum(class_heights[class_num]) / len(class_heights[class_num])
            print(f"{class_num} {average_height}")
        else:  # –ï—Å–ª–∏ –Ω–µ—Ç –¥–∞–Ω–Ω—ã—Ö –¥–ª—è –∫–ª–∞—Å—Å–∞
            print(f"{class_num} -")
